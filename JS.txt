Datatypes in JS:-
Primitive Datatypes:-
+ve,-ve,points also refered as 'number';
Ex:-let a=10.0;
console.log(a);//10.0
console.log(typeof(a));//number

'',"" are used to define string;
Ex:-let b='India';
console.log(b);//India
console.log(typeof(b));//String

true/false are used to define Boolean;
Ex:-let c=true;
console.log(c);//true
console.log(typeof(c));//Boolean

null gives non-primitive case and got o/p as object;[no value]
Ex:-let d=null;
console.log(d);//null
console.log(typeof(d));//object

undefined means not filling any value in the defined variable

console.log(typeof(e));//undefined

Bigint integer means atlast of number give n to get entire number without exponential(+e)[Large values]
Ex:-let f=12564867894121135n;
console.log(f);//12564867894121135n
console.log(typeof(f));//bigint

symbol:-Symbol(' ') is used to display the necessary data to the user;
Ex:-let g=Symbol('Hi');
console.log(g);//Symbol(Hi)
console.log(typeof(g));//symbol

Comparison Operators:-
//<,>,<=,>=,==,===,!=,!==
let x=1;
let y=2;
let z='2';
console.log(x<y);//true
console.log(x>y);//false
console.log(x<=y);//true
console.log(x>=y);//false
console.log(z==y);//true{Checks only Comparison of values neither datatypes}
console.log(z!=y);//false{Checks not Comparison of values neither datatypes}
console.log(z===y)//false{Checks comparison of both values and datatypes}
console.log(z!==y)//true{Checks not comparison of both values and datatypes}

//Logical operators:-
//&&(and) ||(or) !(not)

console.log(z<y || z==y);//true
console.log(!(x<y) && !(z===y));//false
